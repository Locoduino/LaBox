343 int DCCpp::identifyLocoId(volatile RegisterList* inpRegs)
344 {
345  int cv29;
346  cv29 = DCCpp::readCv(inpRegs, 29, 100, 200);
347  if (cv29 == -1)
348  return -1;
349 
350  int id = -1;
351  if (bitRead(cv29, 5))
352  {
353  // long address : get CV#17 and CV#18
354  int cv18 = DCCpp::readCv(inpRegs, 18, 100, 200);
355  if (cv18 != -1)
356  {
357  int cv17 = DCCpp::readCv(inpRegs, 17, 100, 200);
358  if (cv17 != -1)
359  {
360  id = cv18 + ((cv17 - 192) << 8);
361  }
362  }
363  }
364  else
365  {
366  // short address: read only CV#1
367  id = DCCpp::readCv(inpRegs, 1, 100, 200);
368  }
369 
370  return (id);
