34 void TextCommand::receiveCommands()
35 {
36 #ifdef USE_THROTTLES
37  Throttles::loop();
38 #else
39 
40  char c;
41 
42 #if defined(USE_ETHERNET)
43 
44  EthernetClient client = DCCPP_INTERFACE.available();
45 
46  if (client) {
47 
48  if (DCCppConfig::Protocol == EthernetProtocol::HTTP) {
49  DCCPP_INTERFACE.println("HTTP/1.1 200 OK");
50  DCCPP_INTERFACE.println("Content-Type: text/html");
51  DCCPP_INTERFACE.println("Access-Control-Allow-Origin: *");
52  DCCPP_INTERFACE.println("Connection: close");
53  DCCPP_INTERFACE.println("");
54  }
55 
56  while (client.connected() && client.available()) { // while there is data on the network
57  added = Throttle::getCharacter(client.read(), this);
58  c = client.read();
59  if (c == '<') // start of new command
60  commandString[0] = 0;
61  else if (c == '>') // end of new command
62  {
63  MessageStack::MessagesStack.PushMessage(commandString);
64  }
65  else if (strlen(commandString) < MAX_COMMAND_LENGTH) // if comandString still has space, append character just read from network
66  sprintf(commandString, "%s%c", commandString, c); // otherwise, character is ignored (but continue to look for '<' or '>')
67  } // while
68 
69  if (DCCppConfig::Protocol == EthernetProtocol::HTTP)
70  client.stop();
71  }
72 
73 #else // SERIAL case
74 
75  while (DCCPP_INTERFACE.available() > 0) { // while there is data on the serial line
76  c = DCCPP_INTERFACE.read();
77  if (c == '>') // end of new command
78  {
79  MessageStack::MessagesStack.PushMessage(commandString);
80  commandString[0] = 0;
81  }
82  else
83  if (strlen(commandString) < MAX_COMMAND_LENGTH) // if commandString still has space, append character just read from serial line
84  sprintf(commandString, "%s%c", commandString, c); // otherwise, character is ignored (but continue to look for '<' or '>')
85  } // while
86 
87 #endif
88 #endif
89 }

95  void TextCommand::process() {
96 
97 #if !defined(ARDUINO_ARCH_ESP32) || defined(VISUALSTUDIO)
98  // On not ESP32 architectures, load and process immediatly the commands.
99  receiveCommands();
100 #endif
101 
102  int pending = MessageStack::MessagesStack.GetPendingMessageIndex();
103  if (pending != 255)
104  {
105  char bufferStack[MESSAGE_MAXSIZE + 10];
106 
107  bufferStack[0] = 0;
108  MessageStack::MessagesStack.GetMessage(pending, bufferStack);
109 
110  if (bufferStack[0] == 0)
111  return;
112 
113 #ifdef USE_THROTTLES
114  if (bufferStack[0] != '<') // Throttle message
115  {
116  // Throttle message...
117  pCurrentThrottle = Throttle::getThrottleFromStackMessage(bufferStack);
118  if (pCurrentThrottle != NULL)
119  {
120  char buffer[MESSAGE_MAXSIZE + 10];
121  buffer[0] = 0;
122 
123  String message;
124  Throttle::getMessageFromStackMessage(bufferStack, message);
125  strncpy(buffer, message.c_str(), MESSAGE_MAXSIZE - 3);
126 #if defined(DCCPP_DEBUG_MODE)
127  if (buffer[0] != 0)
128  {
129  Serial.print("Throttle ");
130  Serial.print(pCurrentThrottle->getId());
131  Serial.print(" ");
132  Serial.println(buffer);
133  }
134 #endif
135  if (pCurrentThrottle->pConverter != NULL)
136  {
137  pCurrentThrottle->pConverter->convert(pCurrentThrottle, buffer);
138  }
139  else
140  {
141  TextCommand::parse(bufferStack);
142  }
143  }
144  }
145 #endif
146  if (bufferStack[0] == '<') // DCC++ message
147  if (TextCommand::parse(bufferStack + 1) == false)
148  {
149 #if defined(DCCPP_DEBUG_MODE)
150  Serial.println("invalid command !");
151 #endif
152  }
153  }
154  } // TextCommand:process
