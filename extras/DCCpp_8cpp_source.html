516  if (func > 8)

555 int DCCpp::identifyLocoId(volatile RegisterList *inReg)
556 {
557  int id = -1;
558  int temp;
559  temp = inReg->readCVraw(29, 100, 200);
560  if ((temp != -1) && (bitRead(temp, 5))) {
561  // long address : get CV#17 and CV#18
562  id = inReg->readCVraw(18, 100, 200);
563  if (id != -1) {
564  temp = inReg->readCVraw(17, 100, 200);
565  if (temp != -1) {
566  id = id + ((temp - 192) << 8);
567  }
568  }
569  }
570  else {
571  // short address: read only CV#1
572  id = inReg->readCVraw(1, 100, 200);
573  }
574 
575  return (id);
576 }
